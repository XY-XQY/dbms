SELECT c.Cname, a.Acc_Type, a.Amount
FROM Customer c
JOIN Account a ON c.C_Id = a.C_Id
WHERE a.Acc_Type = 'Saving';

ðŸ”¹ 2. Display the data using Natural, Left, and Right join.
-- Natural Join
SELECT * FROM Customer NATURAL JOIN Account;

-- Left Join
SELECT c.C_Id, c.Cname, c.City, a.Acc_Type, a.Amount
FROM Customer c
LEFT JOIN Account a ON c.C_Id = a.C_Id;

-- Right Join
SELECT c.C_Id, c.Cname, c.City, a.Acc_Type, a.Amount
FROM Customer c
RIGHT JOIN Account a ON c.C_Id = a.C_Id;

SELECT Acc_Type, Amount 
FROM Account a
WHERE Amount = (
  SELECT MIN(a2.Amount)
  FROM Account a2
  WHERE a2.Acc_Type = a.Acc_Type
);
